139  template<typename T>
140  bool isPrimitiveType(void) const {
141  if (typeid(T) == typeid(uint8_t) || typeid(T) == typeid(int8_t) ||
142  typeid(T) == typeid(uint16_t) || typeid(T) == typeid(int16_t) ||
143  typeid(T) == typeid(uint32_t) || typeid(T) == typeid(int32_t) ||
144  typeid(T) == typeid(float) || typeid(T) == typeid(double) ||
145  typeid(T) == typeid(uchar) || typeid(T) == typeid(char) ||
146  typeid(T) == typeid(bool)) {
147  return true;
148  } else {
149  return false;
